function unique(arr) {
    let uniqueItems = [];
    for (let i = 0; i < arr.length; i++) {
      element = arr[i];
      let itemArr=[];
      for (let item of arr) {
        itemArr = arr.filter(item => item == element);
        }    
      if (itemArr.length == 1) {
        uniqueItems.push(element)
      }    
      itemArr = itemArr.splice(0, itemArr.length);    
    }
    return uniqueItems;
  }
